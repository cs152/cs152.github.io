def grad(w, X, y):
    xw = linear_function(X, w)
    signed_xw = (2 * y - 1) * xw
    prob_correct = sigmoid(signed_xw)
    dw = -np.dot((1 / prob_correct) * dsigmoid(signed_xw) * (2 * y - 1.), X)
    db = -np.sum((1 / prob_correct) * dsigmoid(signed_xw) * (2 * y - 1.))
