import numpy as np

def matrix_to_latex(matrix):
    """Converts a NumPy matrix to LaTeX."""

    latex_str = "\\begin{pmatrix}\n"
    for row in matrix:
        latex_str += " & ".join(map(str, row)) + "\\\\\n"
    latex_str += "\\end{pmatrix}"
    return latex_str
